#include <stdio.h>
#include <string.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_eth.h"

#include "esp_eth_mac_esp.h"

/* --- p-net Abstraction Headers --- */
#include "pnal_config.h"
#include "pnal_sys.h"
#include "driver_config.h"

/* --- Official p-net API --- */
#include "pnet_api.h"

static const char *TAG = "PROFINET_MOTOR";

/* Hardware Configuration */
#define MY_SPI_HOST      SPI2_HOST
#define PIN_MOSI         23
#define PIN_MISO         32
#define PIN_SCLK         18
#define PIN_CS           5
#define PIN_RST          13 
#define MOTOR_PWM_PIN    14 

/* PROFINET Identifiers (Match GSDML) */
#define API_ID           0
#define SLOT_ID          1
#define SUBSLOT_ID       1

static esp_eth_handle_t s_eth_handle = NULL;
static pnet_t *s_pnet_stack = NULL;
static uint8_t s_mac_addr[6] = {0x02, 0x0A, 0x0B, 0x0C, 0x0D, 0x0E};
static spi_device_handle_t s_ksz_spi_handle;

/* Motor PWM Logic */
void init_motor_pwm(void) {
    ledc_timer_config_t ledc_timer = {
        .speed_mode       = LEDC_LOW_SPEED_MODE,
        .timer_num        = LEDC_TIMER_0,
        .duty_resolution  = LEDC_TIMER_13_BIT,
        .freq_hz          = 50,
        .clk_cfg          = LEDC_AUTO_CLK
    };
    ledc_timer_config(&ledc_timer);

    ledc_channel_config_t ledc_channel = {
        .speed_mode     = LEDC_LOW_SPEED_MODE,
        .channel        = LEDC_CHANNEL_0,
        .timer_sel      = LEDC_TIMER_0,
        .intr_type      = LEDC_INTR_DISABLE,
        .gpio_num       = MOTOR_PWM_PIN,
        .duty           = 410, // ~1ms pulse
        .hpoint         = 0
    };
    ledc_channel_config(&ledc_channel);
}

void set_motor_speed(uint8_t run_cmd, uint8_t speed_cmd) {
    uint32_t target_duty = 410; 
    if (run_cmd == 1) {
        target_duty = 410 + (uint32_t)((speed_cmd / 255.0) * (819 - 410));
    }
    ledc_set_duty(LEDC_LOW_SPEED_MODE, LEDC_CHANNEL_0, target_duty);
    ledc_update_duty(LEDC_LOW_SPEED_MODE, LEDC_CHANNEL_0);
}

/* PROFINET Stack Callbacks */
esp_err_t raw_eth_rx_callback(esp_eth_handle_t eth_handle, uint8_t *buffer, uint32_t length, void *priv) {
    if (buffer != NULL && length > 0) {
        // Feed frame to p-net stack if necessary
        free(buffer); 
    }
    return ESP_OK;
}

int on_pnet_state_change(pnet_t *net, void *arg, uint32_t arep, pnet_event_values_t state) {
    ESP_LOGI(TAG, "PROFINET State: %d", state);
    if (state == PNET_EVENT_PRMEND) {
        pnet_application_ready(net, arep);
    }
    return 0;
}

int on_pnet_exp_module(pnet_t *net, void *arg, uint32_t api, uint16_t slot, uint32_t module_ident) {
    return pnet_plug_module(net, api, slot, module_ident);
}

int on_pnet_exp_submodule(pnet_t *net, void *arg, uint32_t api, uint16_t slot, uint16_t subslot, uint32_t module_ident, uint32_t submodule_ident, const pnet_data_cfg_t *p_exp_data) {
    return pnet_plug_submodule(net, api, slot, subslot, module_ident, submodule_ident, p_exp_data->data_dir, p_exp_data->insize, p_exp_data->outsize);
}

/* Background Task */
void profinet_task(void *pvParameters) {
    uint8_t payload[2] = {0}; 
    uint16_t payload_len = 2;
    uint8_t iops = 0;
    bool new_data = false;

    while (1) {
        if (s_pnet_stack != NULL) {
            pnet_handle_periodic(s_pnet_stack);
            payload_len = sizeof(payload);
            pnet_output_get_data_and_iops(s_pnet_stack, API_ID, SLOT_ID, SUBSLOT_ID, &new_data, payload, &payload_len, &iops);
            if (new_data && iops == PNET_IOXS_GOOD) {
                set_motor_speed(payload[0], payload[1]);
            }
        }
        vTaskDelay(pdMS_TO_TICKS(1)); 
    }
}

void app_main(void) {
    // 1. KSZ8863 Reset
    gpio_set_direction(PIN_RST, GPIO_MODE_OUTPUT);
    gpio_set_level(PIN_RST, 0);
    vTaskDelay(pdMS_TO_TICKS(10));
    gpio_set_level(PIN_RST, 1);
    vTaskDelay(pdMS_TO_TICKS(100));

    init_motor_pwm();

    // 2. SPI Setup
    spi_bus_config_t buscfg = {
        .miso_io_num = PIN_MISO, .mosi_io_num = PIN_MOSI, .sclk_io_num = PIN_SCLK,
        .quadwp_io_num = -1, .quadhd_io_num = -1,
    };
    ESP_ERROR_CHECK(spi_bus_initialize(MY_SPI_HOST, &buscfg, SPI_DMA_CH_AUTO));

    spi_device_interface_config_t spi_devcfg = {
        .clock_speed_hz = 1 * 1000 * 1000, .spics_io_num = PIN_CS, .queue_size = 7
    };
    ESP_ERROR_CHECK(spi_bus_add_device(MY_SPI_HOST, &spi_devcfg, &s_ksz_spi_handle));

    // 3. Ethernet Setup (RMII)
    eth_mac_config_t mac_config = ETH_MAC_DEFAULT_CONFIG();
    eth_esp32_emac_config_t esp32_emac_config = ETH_ESP32_EMAC_DEFAULT_CONFIG();
    esp32_emac_config.interface = EMAC_DATA_INTERFACE_RMII;
    
    esp_eth_mac_t *mac = esp_eth_mac_new_esp32(&esp32_emac_config, &mac_config);
    eth_phy_config_t phy_config = ETH_PHY_DEFAULT_CONFIG();
    phy_config.phy_addr = 1; 
    phy_config.reset_gpio_num = -1; 
    esp_eth_phy_t *phy = esp_eth_phy_new_ksz80xx(&phy_config); 

    esp_eth_config_t eth_config = ETH_DEFAULT_CONFIG(mac, phy);
    eth_config.stack_input = raw_eth_rx_callback; 
    
    ESP_ERROR_CHECK(esp_eth_driver_install(&eth_config, &s_eth_handle));
    esp_eth_ioctl(s_eth_handle, ETH_CMD_S_MAC_ADDR, s_mac_addr);
    ESP_ERROR_CHECK(esp_eth_start(s_eth_handle));

    // 4. Profinet Setup
    pnet_cfg_t pnet_cfg = {0};
    pnet_cfg.tick_us = 1000;
    pnet_cfg.device_id.vendor_id_hi = 0x00;
    pnet_cfg.device_id.vendor_id_lo = 0x25;
    pnet_cfg.device_id.device_id_hi = 0x01;
    pnet_cfg.device_id.device_id_lo = 0x01;
    strncpy(pnet_cfg.station_name, "esp32-motor", PNET_STATION_NAME_MAX_SIZE);
    
    pnet_cfg.state_cb = on_pnet_state_change;
    pnet_cfg.exp_module_cb = on_pnet_exp_module;
    pnet_cfg.exp_submodule_cb = on_pnet_exp_submodule;
    
    s_pnet_stack = pnet_init(&pnet_cfg);
    if (s_pnet_stack) {
        pnet_set_provider_state(s_pnet_stack, true);
        xTaskCreatePinnedToCore(profinet_task, "profinet_task", 8192, NULL, 5, NULL, 1);
    }
}
